import java.util.Scanner;

public class Main {

	public static void main(String[] args) {
		Scanner scn = new Scanner(System.in);
		int len = scn.nextInt();
		String num = scn.next();
		
		int sum=0;
		for(int i=0; i<len; i++) {
			sum += num.charAt(i) - '0';
			
		}
		System.out.println(sum);
	}

	

}
