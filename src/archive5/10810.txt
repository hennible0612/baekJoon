import java.util.Scanner;

public class Main {	
	public static void main(String[] args) {
		Scanner scn = new Scanner(System.in);
		int max = scn.nextInt();

		int array[] = new int[max];
		for(int i=0; i < max; i++) {
			array[i] = scn.nextInt();
		}
		int high = array[0];
		int low = array[0];
		for(int i=0; i< max; i++) {
			if(array[i]> high) {
				high = array[i];
			}
			else if(array[i] < low) {
				low = array[i];
			}
		}
		System.out.print(low + " " + high);
		
		scn.close();
	}
}
