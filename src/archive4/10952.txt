import java.util.Scanner;

public class Main {
	public static void main(String[] args) {
		Scanner scn = new Scanner(System.in);
		
		int A =0;
		int B =0;
		do
		{
			A = scn.nextInt();
			B = scn.nextInt();
			if(A != 0 && B != 0) {
				System.out.println(A + B);
			}
			else
				break;
			
		}while(A!=0 && B!=0);
	}
}
