public class Main {


	public static void main(String[] args) {
		boolean array[] = new boolean[10001];
		for (int i = 1; i < 10001; i++) {
			int n = d(i);
			if (n < 10001) {
				array[n] = true;
			}
		}
		for (int i = 1; i < 10001; i++) {
			if (array[i] == false) {
				System.out.println(i);
			}

		}
	}

	public static int d(int n) {
		int sum = n;
		while (true) {
			sum += n % 10;
			n = n / 10;

			if (n == 0) {
				break;
			}
		}

		return sum;
	}

}
